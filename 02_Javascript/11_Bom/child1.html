const parentColorChanger = document.getElementById("color-btn");
        let isColor = true;
        parentColorChanger.addEventListener("click", () => {
        if (isColor) {
        window.parent.document.body.style.backgroundColor = "aqua"
        isColor = false;
        } else {
        window.parent.document.body.style.backgroundColor = "white"
        isColor = true;
        }
        })
